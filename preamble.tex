%%  Packages that provide the document structure, fonts, and symbols needed to 
%%  write mathematical expressions.

%%  A package that provides theorem-like environments.
\usepackage{amsthm}

%%  A package needed to write bold symbols in math mode.
\usepackage{bm}

%%  A package needed to add color to text.
\usepackage{color}

%%  A package that provides comment environment.
\usepackage{comment}

%%  A package needed to insert images.
\usepackage{graphicx}

%%  A package needed to create hypertext links.
\usepackage{hyperref}
\hypersetup{
    unicode=true, 
    colorlinks=true, 
    citecolor=blue, 
    linkcolor=blue, 
    urlcolor=blue
}

%%  A package needed to write chemical formulea.
\usepackage[version=4]{mhchem}

%%  A packege that provides an environment for colored and framed text boxes.
\usepackage{tcolorbox}
\tcbuselibrary{breakable}

%%  A package that provides advanced drawing capabilities.
\usepackage{tikz}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%    Settings of style                                                   %%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%  Enabling page breaks in math environment
\allowdisplaybreaks[1]

%%  Format of enumerate environment
\renewcommand{\labelenumi}{(\arabic{enumi})}

%%  Numbering
\numberwithin{equation}{section}
\numberwithin{figure}{section}
\numberwithin{table}{section}

%%  Format of footnote
\renewcommand{\thefootnote}{*\arabic{footnote}}

%%  Format of theorem-like environment
\theoremstyle{definition}

%%  Definition environment
\newtheorem{define}{定義}[section]
\newenvironment{definition}{
    \begin{tcolorbox}[
        sharp corners, 
        colback=blue!10, 
        colframe=white, 
        breakable=true
    ]
    \begin{define}
}{
    \end{define}
    \end{tcolorbox}
}

%%  Proposition environment
\newtheorem{prop}{命題}[section]
\newenvironment{proposition}{
    \begin{tcolorbox}[
        sharp corners, 
        colback=red!10, 
        colframe=white, 
        breakable=true
    ]
    \begin{prop}
}{
    \end{prop}
    \end{tcolorbox}
}

%%  Theorem environment
\newtheorem{thm}[prop]{定理}
\newenvironment{theorem}{
    \begin{tcolorbox}[
        sharp corners, 
        colback=red!10, 
        colframe=white, 
        breakable=true
    ]
    \begin{thm}
}{
    \end{thm}
    \end{tcolorbox}
}

%%  Proof environment
\renewcommand{\proofname}{\textgt{証明}}
\renewcommand{\qedsymbol}{$\blacksquare$}
